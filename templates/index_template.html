// 외부 URL일 때만 프록시를 타도록 개선된 변환 함수 async function processImageSource(imgElement) { const src = imgElement.src; // 1. 이미 Base64이거나 로컬 경로(assets/...)인 경우 변환 건너뜀 if (src.startsWith('data:') || !src.startsWith('http')) { return; } // 2. 외부 URL(나무위키 등)인 경우에만 프록시 및 Base64 변환 시도 imgElement.style.opacity = "0"; const b64 = await convertToBase64(src); if (b64) { imgElement.src = b64; imgElement.style.opacity = "1"; } else { imgElement.style.opacity = "0.5"; // 실패 시 반투명 처리 } } // ... (생략) ... document.addEventListener('DOMContentLoaded', () => { // 뱅크 설정 document.querySelectorAll('.band-content').forEach(c => { new Sortable(c, { group: { name: 's', pull: 'clone', put: false }, sort: false }); }); // 드롭 존 설정 document.querySelectorAll('.drop-zone').forEach(z => { new Sortable(z, { group: 's', animation: 150, onAdd: async (e) => { const item = e.item; item.classList.add('tier-item'); const img = item.querySelector('img'); // 수정된 이미지 처리 로직 호출 if (img) { await processImageSource(img); } // 삭제 버튼 추가 (중복 방지) if (!item.querySelector('.del-btn')) { const btn = document.createElement('div'); btn.className = 'del-btn'; btn.innerText = 'X'; btn.onclick = () => item.remove(); item.appendChild(btn); } } }); }); }); async function exportTier() { const n = document.getElementById('tier-capture-area'); const w = document.getElementById('tier-wrapper'); const we = w.classList.contains('expanded'); // 캡처 전 외부 이미지가 남아있다면 최종 변환 const imgs = n.querySelectorAll('img'); for (let i of imgs) { if (i.src.startsWith('http')) { const b64 = await convertToBase64(i.src); if (b64) i.src = b64; } } w.classList.add('expanded'); document.body.classList.add('capturing'); await new Promise(r => setTimeout(r, 600)); domtoimage.toPng(n, { bgcolor: '#fff', // 로컬 이미지 사용 시 품질 향상을 위해 폰트/이미지 로드 대기 시간 확보 }) .then(d => { const l = document.createElement('a'); l.download = `bandori-tier-${Date.now()}.png`; l.href = d; l.click(); }) .catch(err => { console.error("Export Error:", err); }) .finally(() => { document.body.classList.remove('capturing'); if (!we) w.classList.remove('expanded'); }); }
